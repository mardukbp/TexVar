require("tVar/init.lua")
tVar.intString([===[

#Vector matrix multiplication
-- Global Definitions
tVar.outputMode = "RES_EQ"
tVar.decimalSeparator = ","
tMat.eqTexAsMatrix = true
tVar.qOutput = true

-- Variables
M:={{1,12,23},{421,5,6},{733,81,9}}
v:={4,5,6}

-- Calculations
d=(M*v):setName("d"):print():clean()

#Calculate inverse

ans=(M:Inv()*d):setName(""):print()

]===])
